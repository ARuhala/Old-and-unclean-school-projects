yhden mainoksen kok hinta = mainoskampanja.sekuntihinta * mainos.pituus * mainosnayttojenmaara

laskun hinta = mainoskampanjan kaikkien mainosten kok hintojen summa

Mainosnäyttöjen määrä (vaatii mainos_id):
SELECT COUNT(mainos_id) FROM mainosnaytto GROUP BY mainos_id HAVING mainos_id = 4;

Mainoksen pituus (vaatii mainos_id):
SELECT mainos.pituus FROM mainos Where id = 4;

Mainoskampanjan hinnasto (vaatii mainoskampanja_id):
SELECT hinnasto.sekuntihinta FROM mainoskampanja, hinnasto
WHERE mainoskampanja.hinnasto_id = hinnasto.id AND mainoskampanja.id = 1;
